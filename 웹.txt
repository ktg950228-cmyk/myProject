window (최상위, 전역 객체)
├── document   // HTML 문서(DOM 진입점)
│   ├── documentElement (<html>)
│   │   ├── head
│   │   └── body
│   │       ├── h1, p, div ... (실제 HTML 요소)
│   └── getElementById, querySelector ... (DOM 메서드)
├── navigator  // 브라우저 정보
│   ├── userAgent
│   ├── language
│   ├── platform
│   └── onLine
├── location   // URL 제어
│   ├── href
│   ├── pathname
│   ├── protocol
│   └── reload(), assign(), replace()
├── history    // 방문 기록
│   ├── length
│   └── back(), forward(), go()
├── screen     // 모니터 정보
│   ├── width
│   ├── height
│   ├── availWidth
│   └── availHeight
└── alert(), prompt(), setTimeout() ... (window 전역 함수)
└──사용자 정의 객체/변수
